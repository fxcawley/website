def lasso_regression_with_constraints(X, y, max_features):
    n_samples, n_features = X.shape
    # Standardize features
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(X)
    y_scaled = y.values

    # Define the optimization variables
    beta = cp.Variable(n_features)
    lambd = cp.Parameter(nonneg=True)

    # Lasso objective function
    objective = cp.Minimize((1 / n_samples) * cp.sum_squares(X_scaled @ beta - y_scaled) + lambd * cp.norm1(beta))

    # Non-negativity constraints
    constraints = [beta >= 0]

    # Problem definition
    problem = cp.Problem(objective, constraints)

    # Iteratively increase lambda until we have at most max_features non-zero coefficients
    lambda_value = 0.001
    max_lambda = 1000
    tol = 1e-4
    selected = []
    while lambda_value < max_lambda:
        lambd.value = lambda_value
        problem.solve(solver=cp.SCS)
        beta_value = beta.value
        selected = np.where(beta_value > tol)[0]
        if len(selected) <= max_features:
            break
        lambda_value *= 1.5  # Increase lambda
    else:
        print("Warning: Couldn't reduce features to desired number.")

    # Retrieve the weights
    weights = pd.Series(beta_value, index=X.columns)
    # Keep only non-zero weights
    weights = weights[weights > tol]

    return weights

# Calculate performance metrics
def performance_metrics(actual_returns, portfolio_returns):
    # Correlation
    corr = np.corrcoef(actual_returns, portfolio_returns)[0,1]
    # Tracking Error
    tracking_error = np.std(actual_returns - portfolio_returns) * np.sqrt(252)
    # Cumulative Returns
    cumulative_actual = (1 + actual_returns).cumprod() - 1
    cumulative_portfolio = (1 + portfolio_returns).cumprod() - 1
    total_return_actual = cumulative_actual.iloc[-1]
    total_return_portfolio = cumulative_portfolio.iloc[-1]
    # Volatility
    vol_actual = np.std(actual_returns) * np.sqrt(252)
    vol_portfolio = np.std(portfolio_returns) * np.sqrt(252)
    # Sharpe Ratio (Assuming risk-free rate is zero)
    sharpe_actual = np.mean(actual_returns) / np.std(actual_returns) * np.sqrt(252)
    sharpe_portfolio = np.mean(portfolio_returns) / np.std(portfolio_returns) * np.sqrt(252)

    metrics = {
        'Correlation': corr,
        'Tracking Error': tracking_error,
        'Total Return (Stock)': total_return_actual,
        'Total Return (Portfolio)': total_return_portfolio,
        'Volatility (Stock)': vol_actual,
        'Volatility (Portfolio)': vol_portfolio,
        'Sharpe Ratio (Stock)': sharpe_actual,
        'Sharpe Ratio (Portfolio)': sharpe_portfolio
    }

    return metrics

def on_stock_change(change):
    if change['type'] == 'change' and change['name'] == 'value':
        selected_stock = change['new']
        print(f"\nYou selected: {selected_stock}")

        # Data retrieval
        in_sample_start = '2022-01-01'
        in_sample_end = '2022-12-31'
        out_sample_start = '2023-01-01'
        out_sample_end = '2023-10-31'  # Adjust based on current date

        in_sample_data = get_data(selected_stock, etf_list, in_sample_start, in_sample_end)
        in_sample_returns = in_sample_data.pct_change().dropna()

        out_sample_data = get_data(selected_stock, etf_list, out_sample_start, out_sample_end)
        out_sample_returns = out_sample_data.pct_change().dropna()

        # Correlation computation
        correlations = in_sample_returns.corr()[selected_stock].drop(selected_stock)
        top_pos_corr = correlations.nlargest(10)
        top_neg_corr = correlations.nsmallest(10)
        selected_etfs = list(top_pos_corr.index) + list(top_neg_corr.index)
        X_in_sample = in_sample_returns[selected_etfs]
        y_in_sample = in_sample_returns[selected_stock]

        # Model fitting
        max_etfs = 5
        weights = lasso_regression_with_constraints(X_in_sample, y_in_sample, max_etfs)
        print("\nSelected ETFs and Weights:")
        print(weights)

        # In-sample performance
        portfolio_in_sample_returns = X_in_sample[weights.index] @ weights.values
        in_sample_metrics = performance_metrics(y_in_sample, portfolio_in_sample_returns)
        print("\nIn-Sample Performance Metrics:")
        for key, value in in_sample_metrics.items():
            print(f"{key}: {value:.4f}")

        # Out-of-sample performance
        X_out_sample = out_sample_returns[weights.index]
        y_out_sample = out_sample_returns[selected_stock]
        portfolio_out_sample_returns = X_out_sample @ weights.values
        out_sample_metrics = performance_metrics(y_out_sample, portfolio_out_sample_returns)
        print("\nOut-of-Sample Performance Metrics:")
        for key, value in out_sample_metrics.items():
            print(f"{key}: {value:.4f}")

        # Visualization
        cumulative_stock_in = (1 + y_in_sample).cumprod() - 1
        cumulative_portfolio_in = (1 + portfolio_in_sample_returns).cumprod() - 1
        cumulative_stock_out = (1 + y_out_sample).cumprod() - 1
        cumulative_portfolio_out = (1 + portfolio_out_sample_returns).cumprod() - 1
